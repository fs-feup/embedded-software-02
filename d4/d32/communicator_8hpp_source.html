160inline void Communicator::res_state_callback(const uint8_t *buf) {
161 bool emg_stop1 = buf[0] & 0x01;
162 bool emg_stop2 = buf[3] >> 7 & 0x01;
163 bool go_switch = (buf[0] >> 1) & 0x01;
164 bool go_button = (buf[0] >> 2) & 0x01;
165
166 if (go_button || go_switch)
167 _systemData->r2d_logics_.process_go_signal();
168 else if (!emg_stop1 && !emg_stop2) {
169 _systemData->failure_detection_.emergency_signal_ = true;
170 }
171
172 _systemData->failure_detection_.radio_quality_ = buf[6];
173 bool signal_loss = (buf[7] >> 6) & 0x01;
174 if (!signal_loss) {
175 _systemData->failure_detection_.res_signal_loss_timestamp_

192inline void Communicator::bamocar_callback(const uint8_t *buf) {
193 _systemData->failure_detection_.inversor_alive_timestamp_.reset();
194
195 if (buf[0] == BTB_READY) {
196 if (buf[1] == false) {
197 _systemData->failure_detection_.ts_on_ = false;
198 }
199 } else if (buf[0] == VDC_BUS) {
200 unsigned dc_voltage = (buf[2] << 8) | buf[1];
201 _systemData->failure_detection_.dc_voltage_ = dc_voltage;
202
203 if (dc_voltage < DC_THRESHOLD) {
204 _systemData->failure_detection_.dc_voltage_hold_timestamp_.reset();
205 if (_systemData->failure_detection_.dc_voltage_drop_timestamp_.checkWithoutReset()) {
206 _systemData->failure_detection_.ts_on_ = false;
207 }
208 } else {
209 _systemData->failure_detection_.dc_voltage_drop_timestamp_.reset();
210 if (_systemData->failure_detection_.dc_voltage_hold_timestamp_.checkWithoutReset()) {
211 _systemData->failure_detection_.ts_on_ = true;
212 }
213 }

231inline void Communicator::parse_message(const CAN_message_t &msg) {
232 switch (msg.id) {
233 case AS_CU_ID:
234 pc_callback(msg.buf);
235 case RES_STATE:
236 res_state_callback(msg.buf);
237 break;
238 case RES_READY:
239 res_ready_callback();
240 break;
241 case C1_ID:
242 c1_callback(msg.buf); // rwheel, lwheel and hydraulic line
243 break;
244 case BAMO_RESPONSE_ID:
245 bamocar_callback(msg.buf);
246 break;
247 case STEERING_ID:
248 steering_callback();
249 break;
250 default:
251 break;
252 }
253}
