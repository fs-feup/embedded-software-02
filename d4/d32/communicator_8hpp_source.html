152 bool emg_stop1 = buf[0] & 0x01;
153 bool emg_stop2 = buf[3] >> 7 & 0x01;
154 bool go_switch = (buf[0] >> 1) & 0x01;
155 bool go_button = (buf[0] >> 2) & 0x01;
156
157 if (go_button || go_switch)
158 _systemData->r2d_logics_.process_go_signal();

182inline void Communicator::bamocar_callback(const uint8_t *buf) {
183 _systemData->failure_detection_.inversor_alive_timestamp_.reset();
184
185 if (buf[0] == BTB_READY) {
186 if (buf[1] == false) {
187 _systemData->failure_detection_.ts_on_ = false;
188 }
189 } else if (buf[0] == BAMOCAR_BATTERY_VOLTAGE_CODE) {
190 unsigned dc_voltage = (buf[2] << 8) | buf[1];
191 _systemData->failure_detection_.dc_voltage_ = dc_voltage;
192
193 // Voltage hysteresis:
194 if (dc_voltage < DC_THRESHOLD) {
195 // When voltage drops/is below threshold:
196 // Reset hold timer and check if voltage has been below threshold long enough
197 _systemData->failure_detection_.dc_voltage_hold_timestamp_.reset();
198 if (_systemData->failure_detection_.dc_voltage_drop_timestamp_.checkWithoutReset()) {
199 _systemData->failure_detection_.ts_on_ = false;
200 }
201 } else {
202 // When voltage is above threshold:
203 // Reset drop timer and check if voltage has been above threshold long enough
204 _systemData->failure_detection_.dc_voltage_drop_timestamp_.reset();
205 if (_systemData->failure_detection_.dc_voltage_hold_timestamp_.checkWithoutReset()) {
206 _systemData->failure_detection_.ts_on_ = true;
207 }
208 }
209 }

226inline void Communicator::parse_message(const CAN_message_t &msg) {
227 switch (msg.id) {
228 case AS_CU_ID:
229 pc_callback(msg.buf);
230 case RES_STATE:
231 res_state_callback(msg.buf);
232 break;
233 case RES_READY:
234 res_ready_callback();
235 break;
236 case BAMO_RESPONSE_ID:
237 bamocar_callback(msg.buf);
238 break;
239 case STEERING_ID:
240 steering_callback();
241 break;
242 default:
243 break;
244 }
