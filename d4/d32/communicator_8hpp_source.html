191inline void Communicator::bamocar_callback(const uint8_t *buf) {
192 _systemData->updatable_timestamps_.inversor_alive_timestamp_.reset();
193
194 if (buf[0] == BTB_READY) {
195 if (buf[1] == false) {
196 _systemData->failure_detection_.ts_on_ = false;
197 }
198 } else if (buf[0] == BAMOCAR_BATTERY_VOLTAGE_CODE) {
199 unsigned dc_voltage = (buf[2] << 8) | buf[1];
200 _systemData->failure_detection_.dc_voltage_ = dc_voltage;
201
202 // Voltage hysteresis:
203 if (dc_voltage < DC_THRESHOLD) {
204 // When voltage drops/is below threshold:
205 // Reset hold timer and check if voltage has been below threshold long enough
206 _systemData->updatable_timestamps_.dc_voltage_hold_timestamp_.reset();
207 if (_systemData->updatable_timestamps_.dc_voltage_drop_timestamp_.checkWithoutReset()) {
208 _systemData->failure_detection_.ts_on_ = false;
209 }
210 } else {
211 // When voltage is above threshold:
212 // Reset drop timer and check if voltage has been above threshold long enough
213 _systemData->updatable_timestamps_.dc_voltage_drop_timestamp_.reset();
214 if (_systemData->updatable_timestamps_.dc_voltage_hold_timestamp_.checkWithoutReset()) {
215 _systemData->failure_detection_.ts_on_ = true;
216 }
217 }
218 }

244inline void Communicator::parse_message(const CAN_message_t &msg) {
245 switch (msg.id) {
246 case AS_CU_ID:
247 pc_callback(msg.buf);
248 case RES_STATE:
249 res_state_callback(msg.buf);
250 break;
251 case RES_READY:
252 res_ready_callback();
253 break;
254 case BAMO_RESPONSE_ID:
255 bamocar_callback(msg.buf);
256 break;
257 case STEERING_ID:
258 steering_callback();
259 break;
260 case DASH_ID:
261 dash_callback(msg.buf);
262 break;
263 default:
264 break;
265 }
266}
