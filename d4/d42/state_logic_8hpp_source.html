56inline void ASState::calculate_state() {
57 switch (state_) {
58 case State::AS_MANUAL:
59 if (_checkup_manager_.should_stay_manual_driving()) {
60 break;
61 }
62
63 DEBUG_PRINT("Entering OFF state from MANUAL");
64 _output_coordinator_->enter_off_state();
65
66 state_ = State::AS_OFF;
67 break;
68
69 case State::AS_OFF:
70
71 if (!timer_has_started) {
72 emergency_timer_.begin(
73 [] {
74 instance->timer_started();
75 if (instance->_checkup_manager_.should_enter_emergency(instance->state_)) {
76 instance->_output_coordinator_->enter_emergency_state();
77 instance->_checkup_manager_._ebs_sound_timestamp_.reset();
78 instance->state_ = State::AS_EMERGENCY;
79 }
80 },
81 150'000);
82 }
83 // If manual driving checkup fails, the car can't be in OFF state, so it goes back to MANUAL
84 if (_checkup_manager_.should_stay_manual_driving()) {
85 DEBUG_PRINT("Entering MANUAL state from OFF");
86 _output_coordinator_->enter_manual_state();
87 state_ = State::AS_MANUAL;
88 break;
89 }
90 _output_coordinator_->refresh_r2d_vars();
91 _output_coordinator_->refresh_emergency_vars();
92 if (_checkup_manager_.should_stay_off()) break;
93 if (!_checkup_manager_.should_go_ready_from_off()) break; // recheck all states
94
95 DEBUG_PRINT("Entering READY state from OFF");
96 _output_coordinator_->enter_ready_state();
97 state_ = State::AS_READY;
98 DEBUG_PRINT("READY state entered...");
99 break;
100
101 case State::AS_READY:
102
103 if (_checkup_manager_.should_stay_ready()) {
104 break;
105 }
106 _output_coordinator_->enter_driving_state();
107 state_ = State::AS_DRIVING;
108 break;
109
110 case State::AS_DRIVING:
111 _output_coordinator_->blink_driving_led();
112 if (_checkup_manager_.should_stay_driving()) break;
113 _output_coordinator_->enter_finish_state();
114 state_ = State::AS_FINISHED;
115 break;
116
117 case State::AS_FINISHED:
118 if (_checkup_manager_.res_triggered()) {
119 DEBUG_PRINT("Entering EMERGENCY state from FINISHED");
120
121 _output_coordinator_->enter_emergency_state();
122 _checkup_manager_._ebs_sound_timestamp_.reset();
123 state_ = State::AS_EMERGENCY;
124 break;
125 }
126 if (_checkup_manager_.should_stay_mission_finished()) break;
127
128 DEBUG_PRINT("Entering OFF state from FINISHED");
129 _output_coordinator_->enter_off_state();
130 _checkup_manager_.reset_checkup_state();
131 state_ = State::AS_OFF;
132 break;
133
134 case State::AS_EMERGENCY:
135 _output_coordinator_->blink_emergency_led();
136
137 if (_checkup_manager_.emergency_sequence_complete()) {
138 DEBUG_PRINT("Entering OFF state from EMERGENCY");
139 _output_coordinator_->enter_off_state();
140 _checkup_manager_.reset_checkup_state();
141
142 state_ = State::AS_OFF;
143 break;
144 }
145 break;
146 default:
147 break;
148 }
