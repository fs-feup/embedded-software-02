189inline void DigitalReceiver::read_mission() {
190 const int raw_value = analogRead(AMI);
191 int mapped_value = map(constrain(raw_value, 0, ADC_MAX_VALUE), 0, ADC_MAX_VALUE, 0, MAX_MISSION); // constrain just in case
192 Mission latest_mission = static_cast<Mission>(mapped_value);
193
194 if ((latest_mission == *mission_) && (latest_mission == last_tried_mission_)) {
195 mission_change_counter_ = 0;
196 } else {
197 mission_change_counter_++;
198 }
199 this->last_tried_mission_ = latest_mission;
200 if (mission_change_counter_ >= CHANGE_COUNTER_LIMIT) {
201 *mission_ = latest_mission;
202 mission_change_counter_ = 0;
203 }
204 *mission_ = Mission::MANUAL;
205}

229inline void DigitalReceiver::read_rpm() {
230 unsigned long time_interval_rr = (last_wheel_pulse_rr - second_to_last_wheel_pulse_rr);
231 unsigned long time_interval_rl = (last_wheel_pulse_rl - second_to_last_wheel_pulse_rl);
232 if (micros() - last_wheel_pulse_rr > LIMIT_RPM_INTERVAL) {
233 hardware_data_->rr_wheel_rpm = 0.0;
234 } else {
235 hardware_data_->rr_wheel_rpm =
236 1 / (time_interval_rr * MICRO_TO_SECONDS * PULSES_PER_ROTATION) * SECONDS_IN_MINUTE;
237 }
238 if (micros() - last_wheel_pulse_rl > LIMIT_RPM_INTERVAL) {
239 hardware_data_->rl_wheel_rpm = 0.0;
240 } else {
241 hardware_data_->rl_wheel_rpm =
242 1 / (time_interval_rl * MICRO_TO_SECONDS * PULSES_PER_ROTATION) * SECONDS_IN_MINUTE;
243 }
