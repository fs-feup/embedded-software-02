221inline void DigitalReceiver::read_mission() {
222 int raw_value = analogRead(AMI);
223 int mapped_value = map(constrain(raw_value, 0, ADC_MAX_VALUE), 0, ADC_MAX_VALUE, 0, MAX_MISSION); // constrain just in case
224 Mission latest_mission = static_cast<Mission>(mapped_value);
225
226 if ((latest_mission == system_data_->mission_) && (latest_mission == last_tried_mission_)) {
227 mission_change_counter_ = 0;
228 } else {
229 mission_change_counter_++;
230 }
231 this->last_tried_mission_ = latest_mission;
232 if (mission_change_counter_ >= CHANGE_COUNTER_LIMIT) {
233 system_data_->mission_ = latest_mission;
234 mission_change_counter_ = 0;
235 }
236}

259inline void DigitalReceiver::read_rpm() {
260 unsigned long time_interval_rr = (last_wheel_pulse_rr - second_to_last_wheel_pulse_rr);
261 unsigned long time_interval_rl = (last_wheel_pulse_rl - second_to_last_wheel_pulse_rl);
262 if (micros() - last_wheel_pulse_rr > LIMIT_RPM_INTERVAL) {
263 system_data_->hardware_data_._right_wheel_rpm = 0.0;
264 } else {
265 system_data_->hardware_data_._right_wheel_rpm =
266 1 / (time_interval_rr * MICRO_TO_SECONDS * PULSES_PER_ROTATION) * SECONDS_IN_MINUTE;
267 }
268 if (micros() - last_wheel_pulse_rl > LIMIT_RPM_INTERVAL) {
269 system_data_->hardware_data_._left_wheel_rpm = 0.0;
270 } else {
271 system_data_->hardware_data_._left_wheel_rpm =
272 1 / (time_interval_rl * MICRO_TO_SECONDS * PULSES_PER_ROTATION) * SECONDS_IN_MINUTE;
273 }
