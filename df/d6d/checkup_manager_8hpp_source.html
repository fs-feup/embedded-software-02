183inline CheckupManager::CheckupError CheckupManager::initial_checkup_sequence() {
184 switch (checkup_state_) {
185 case CheckupState::WAIT_FOR_ASMS:
186 if (_system_data_->hardware_data_.asms_on_) {
187 checkup_state_ = CheckupState::CHECK_BRAKE_PRESSURE;
188 DEBUG_PRINT("ASMS activated, starting watchdog check");
189 }
190 break;
191 case CheckupState::START_TOGGLING_WATCHDOG:
192 if (_system_data_->hardware_data_.wd_ready_) {
193 checkup_state_ = CheckupState::TOGGLING_WATCHDOG;
194 _watchdog_toggle_timer_.reset();
195 DEBUG_PRINT("Watchdog ready, starting toggle sequence");
196 break;
197 }
198 DigitalSender::toggle_watchdog();
199 break;
200
201 case CheckupState::TOGGLING_WATCHDOG:
202 // Fail immediately if WD_READY goes low during toggling
203 if (!_system_data_->hardware_data_.wd_ready_) {
204 DEBUG_PRINT("Watchdog error: WD_READY went low during toggling phase");
205 return CheckupError::ERROR_WD_TOGGLE;
206 }
207
208 DigitalSender::toggle_watchdog();
209
210 // Toggle for the specified duration
211 if (_watchdog_toggle_timer_.checkWithoutReset()) {
212 checkup_state_ = CheckupState::STOP_TOGGLING_WATCHDOG;
213 _watchdog_test_timer_.reset();
214 DEBUG_PRINT("Watchdog toggle complete, stopping watchdog");
215 }
216 break;
217
218 case CheckupState::STOP_TOGGLING_WATCHDOG:
219 _watchdog_test_timer_.reset();
220 checkup_state_ = CheckupState::CHECK_WATCHDOG;
221 DEBUG_PRINT("Stopping watchdog toggle, beginning verification");
222 break;
223
224 case CheckupState::CHECK_WATCHDOG:
225 if (!_system_data_->hardware_data_.wd_ready_) {
226 DigitalSender::close_watchdog_sdc();
227
228 checkup_state_ = CheckupState::START_TOGGLING_WATCHDOG_AGAIN;
229 DEBUG_PRINT("Watchdog no longer ready - verification successful");
230 break;
231 }
232 if (_watchdog_test_timer_.checkWithoutReset()) {
233 DEBUG_PRINT("Watchdog test failed - WD_READY did not go low during verification time");
234 return CheckupError::ERROR_WD_STAYED_READY;
235 }
236 break;
237 case CheckupState::START_TOGGLING_WATCHDOG_AGAIN:
238 watchdog_timer_.begin([] { DigitalSender::toggle_watchdog(); }, 50'000);
239 checkup_state_ = CheckupState::CHECK_EBS_STORAGE;
240 break;
241 case CheckupState::CHECK_EBS_STORAGE:
242 DEBUG_PRINT("EBS Storage - pressure: " + String(_system_data_->hardware_data_.pneumatic_line_pressure_));
243 if (_system_data_->hardware_data_.pneumatic_line_pressure_) {
244 checkup_state_ = CheckupState::CHECK_BRAKE_PRESSURE;
245 }
246 break;
247 case CheckupState::CHECK_BRAKE_PRESSURE:
248 DEBUG_PRINT("Hydraulic Pressure: " + String(_system_data_->hardware_data_._hydraulic_line_pressure) + " - " +
249 String(_system_data_->hardware_data_.hydraulic_line_front_pressure));
250 if (_system_data_->hardware_data_._hydraulic_line_pressure >= HYDRAULIC_BRAKE_THRESHOLD &&
251 _system_data_->hardware_data_.hydraulic_line_front_pressure >= HYDRAULIC_BRAKE_THRESHOLD) {
252 checkup_state_ = CheckupState::WAIT_FOR_ASATS;
253 }
254 break;
255 case CheckupState::WAIT_FOR_ASATS:
256 if (_system_data_->hardware_data_.asats_pressed_) {
257 _system_data_->failure_detection_.emergency_signal_ = false;
258 checkup_state_ = CheckupState::CLOSE_SDC;
259 DEBUG_PRINT("AS ATS Pressed");
260 }
261 break;
262 case CheckupState::CHECK_TIMESTAMPS: {
263 DEBUG_PRINT("Checking timestamps");
264 if (_system_data_->failure_detection_.has_any_component_timed_out() ||
265 _system_data_->failure_detection_.emergency_signal_) {
266 return CheckupError::ERROR_TIMESTAMPS_EMERGENCY;
267 }
268 checkup_state_ = CheckupState::CLOSE_SDC;
269 break;
270 }
271 case CheckupState::CLOSE_SDC:
272 DigitalSender::close_sdc();
273 checkup_state_ = CheckupState::WAIT_FOR_TS;
274 DEBUG_PRINT("Closing SDC");
275 break;
276 case CheckupState::WAIT_FOR_TS:
277 DEBUG_PRINT("Waiting for TS activation");
278 if (_system_data_->failure_detection_.ts_on_) {
279 DEBUG_PRINT("TS activated");
280 checkup_state_ = CheckupState::EBS_CHECKS;
281 }
282 break;
283 case CheckupState::EBS_CHECKS:
284 DEBUG_PRINT("Starting EBS checks");
285 handle_ebs_check();
286 break;
287 case CheckupState::CHECKUP_COMPLETE:
288 // Final state, all checks passed
289 // DEBUG_PRINT("Checkup complete, transitioning to ready state");
290 return CheckupError::SUCCESS;
291
292
293 default:
294 break;
295 }
296 return CheckupError::WAITING_FOR_RESPONSE;
297}

299inline void CheckupManager::handle_ebs_check() {
300 switch (pressure_test_phase_) {
301 case EbsPressureTestPhase::DISABLE_ACTUATOR_1:
302 // Step 10: Disable EBS actuator 1
303 DEBUG_PRINT("Disabling EBS actuator 1");
304 DigitalSender::disable_ebs_actuator_1();
305 pressure_test_phase_ = EbsPressureTestPhase::CHANGE_ACTUATORS;
306 break;
307
308 case EbsPressureTestPhase::CHECK_ACTUATOR_2:
309 if (_system_data_->hardware_data_.pneumatic_line_pressure_ &&
310 _system_data_->hardware_data_.hydraulic_line_front_pressure < HYDRAULIC_BRAKE_THRESHOLD &&
311 _system_data_->hardware_data_._hydraulic_line_pressure >= HYDRAULIC_BRAKE_THRESHOLD) {//pressure should be high even with only one actuator
312 DEBUG_PRINT("Pressure high confirmed with only actuator 2");
313 pressure_test_phase_ = EbsPressureTestPhase::CHANGE_ACTUATORS;
314 }
315 break;
316
317 case EbsPressureTestPhase::CHANGE_ACTUATORS:
318 // Step 12: Enable EBS actuator 1 again and disable actuator 2
319 DEBUG_PRINT("Re-enabling EBS actuator 1 and disabling actuator 2");
320 DigitalSender::enable_ebs_actuator_1();
321 DigitalSender::disable_ebs_actuator_2();
322 pressure_test_phase_ = EbsPressureTestPhase::ENABLE_ACTUATOR_2;
323 break;
324
325 case EbsPressureTestPhase::CHECK_ACTUATOR_1:
326 // Step 13: Check that the brake pressure is still built up correctly
327 if (_system_data_->hardware_data_.pneumatic_line_pressure_ &&
328 _system_data_->hardware_data_.hydraulic_line_front_pressure >= HYDRAULIC_BRAKE_THRESHOLD &&
329 _system_data_->hardware_data_._hydraulic_line_pressure < HYDRAULIC_BRAKE_THRESHOLD) {
330 DEBUG_PRINT("Pressure high confirmed with only actuator 1");
331 pressure_test_phase_ = EbsPressureTestPhase::ENABLE_ACTUATOR_2;
332 }
333 break;
334
335 case EbsPressureTestPhase::ENABLE_ACTUATOR_2:
336 // Step 14: Enable EBS actuator 2 again
337 DEBUG_PRINT("Re-enabling EBS actuator 2");
338 DigitalSender::enable_ebs_actuator_2();
339 pressure_test_phase_ = EbsPressureTestPhase::COMPLETE;
340 break;
341
342 case EbsPressureTestPhase::CHECK_BOTH_ACTUATORS:
343 // Step 15: Check that both actuators are working correctly
344 DEBUG_PRINT("Final pressure check");
345 if (_system_data_->hardware_data_.pneumatic_line_pressure_ &&
346 _system_data_->hardware_data_.hydraulic_line_front_pressure >= HYDRAULIC_BRAKE_THRESHOLD &&
347 _system_data_->hardware_data_._hydraulic_line_pressure >= HYDRAULIC_BRAKE_THRESHOLD) {
348 DEBUG_PRINT("Both actuators confirmed working correctly");
349 pressure_test_phase_ = EbsPressureTestPhase::COMPLETE;
350 }
351 break;
352
353 case EbsPressureTestPhase::COMPLETE:
354 // Step 16: Transition to ready state
355 DEBUG_PRINT("EBS check complete, transitioning to next state");
356 checkup_state_ = CheckupState::CHECKUP_COMPLETE;
357 pressure_test_phase_ = EbsPressureTestPhase::DISABLE_ACTUATOR_1;
358 break;
359 }
360}
